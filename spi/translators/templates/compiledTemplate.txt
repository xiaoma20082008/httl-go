package templates

import (
	"github.com/xiaoma20082008/httl-go"
	"io"
)

type ${name}CompiledTemplate struct {
	BaseTemplate
}

func (t *${name}CompiledTemplate) Evaluate(o map[string]any) (string, error) {
	sw := StringWriter{}
	defer sw.Close()
	if t.Render(o, &sw) != nil {
		return "", nil
	}
	return sw.ToString(), nil
}

func (t *${name}CompiledTemplate) Render(o map[string]any, w io.Writer) error {
	v := CompiledVisitor{}
	return t.Accept(&v)
}

func New${name}CompiledTemplate httl.Template {
	return &${name}CompiledTemplate{}
}
